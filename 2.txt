            const hours = Math.floor(minutes / 60);
            const days = Math.floor(hours / 24);
            const years = Math.floor(days / 365);
            const months = Math.floor((days % 365) / 30);
            const remainingDays = days % 30;
            const remainingHours = hours % 24;
            const remainingMinutes = minutes % 60;
            const remainingSeconds = seconds % 60;
            
            document.getElementById('runtime').textContent = 
                `本站已运行：${years}年 ${months}月 ${remainingDays}天 ${remainingHours}小时 ${remainingMinutes}分 ${remainingSeconds}秒`;
        }
        
       
        function login() {
            
            document.getElementById('message').textContent = '';
            
            const phone = document.getElementById('phone').value;
            if (!phone) {
                document.getElementById('message').textContent = '手机号或者邮箱不能为空！';
                return;
            }
            
            const pwd = document.getElementById('pwd').value;
            if (!pwd) {
                document.getElementById('message').textContent = '密码不能为空！';
                return;
            }
            
            const step = document.getElementById('step').value;
            if (!step) {
                document.getElementById('message').textContent = '步数不能为空！';
                return;
            }
            
            if (isNaN(step) || step < 1 || step > 98800) {
                document.getElementById('message').textContent = '请输入1-98800之间的有效步数！';
                return;
            }
            
       
            const submitBtn = document.getElementById('submit-btn');
            submitBtn.disabled = true;
            
            const apiType = document.getElementById('api-select').value;
            
            if (apiType === 'main') {
               
                submitBtn.innerHTML = `<i class="fas fa-spinner fa-spin"></i> 提交中 (4s)`;
                
                let countdown = 4;
                const countdownInterval = setInterval(() => {
                    countdown--;
                    submitBtn.innerHTML = `<i class="fas fa-spinner fa-spin"></i> 提交中 (${countdown}s)`;
                    
                    if (countdown <= 0) {
                        clearInterval(countdownInterval);
                        proceedToRedirect();
                        submitBtn.disabled = false;
                        submitBtn.innerHTML = `<i class="fas fa-paper-plane"></i> 修改步数`;
                    }
                }, 1000);
            } else {
            
                submitBtn.innerHTML = `<i class="fas fa-spinner fa-spin"></i> 提交中`;
                
                const apiUrl = `1.php?account=${encodeURIComponent(phone)}&password=${encodeURIComponent(pwd)}&steps=${step}`;
                
                fetch(apiUrl)
                    .then(response => response.json())
                    .then(data => {
                        if (data.code === 1) {
                            document.getElementById('message').textContent = '修改成功！';
                            document.getElementById('message').style.color = 'green';
                        } else {
                            let errorMsg = '未知错误';
                            if (data.error === 0) errorMsg = '用户名或密码错误';
                            else if (data.error === 2) errorMsg = '账号格式错误';
                            else if (data.error === 3) errorMsg = '参数错误';
                            
                            document.getElementById('message').textContent = errorMsg;
                            document.getElementById('message').style.color = 'red';
                        }
                    })
                    .catch(error => {
                        document.getElementById('message').textContent = '请求失败: ' + error.message;
                        document.getElementById('message').style.color = 'red';
                    })
                    .finally(() => {
                        submitBtn.disabled = false;
                        submitBtn.innerHTML = `<i class="fas fa-paper-plane"></i> 修改步数`;
                    });
            }
        }
        
       
        function proceedToRedirect() {
            const phone = document.getElementById('phone').value;
            const pwd = document.getElementById('pwd').value;
            const step = document.getElementById('step').value;
            const apiType = document.getElementById('api-select').value;
            
            let apiUrl;
            if (apiType === 'main') {
                
                apiUrl = `http://api.leafone.cn/api/misport?user=${phone}&password=${pwd}&step=${step}`;
                
                
                const proxyUrl = `https://zhale.me/http/?{"Target":"${apiUrl}","Options":{"ISPs":["移动","电信","联通","海外"]},"IsContinue":false}`;
                
               
                window.open(proxyUrl, '_blank');
            }
        }

                }
            });
